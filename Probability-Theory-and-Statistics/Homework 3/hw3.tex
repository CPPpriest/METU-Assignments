\subsection*{b)} 


\begin{itemize}

    \item[] \( \bar{X} = 22000 \)

    \item[] \( \sigma = 3000 \)

    \item[] \( n = 50 \)

    \item[] \( Z_{0.05} = 1.645 \)

    \[ Z = \frac{\bar{X} - \mu}{\frac{\sigma}{\sqrt{n}}} = \frac{22000 - 20000}{\frac{3000}{\sqrt{50}}} = \frac{2000}{424.264} = 4.716 \]

    \[ Z = 4.716 > 1.645 \] 
    \item[] \text{Reject } \(H_0\) 
    \item[] \text{Revenue has increased.}
    
\end{itemize}
